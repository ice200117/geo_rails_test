def self.history_data(city, start_time)
        d = city.china_cities_hours.group_by_day(:data_real_time, range: start_time..Time.now).average(:AQI)
        hs = {}
        d.each do |k,v|
                hs[[city.city_name+'监测值', k.strftime("%d%b")]] = v.round
        end
        hs
end
